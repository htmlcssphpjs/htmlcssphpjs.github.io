$LIB

Библиотека для работы с DOM

$.js('js/script.js');

$lib

  • Лёгкая настройка
  • Всё открыто и документировано
  • И мы всегда готовы выслушать

Подключение

<script src="lib.js"></script>

И скрипт готов к работе

npm

Установка через npm:

npm install htmlcssphpjs/libs

Использование

Все команды(функции) начинаются с $, например:

let img = $.find('.image');

Вот код для тестирования:

  1.   //найти эллемент
  2.   let logo = $.find('#logo');
  3.   //удалить эллемент
  4.   $.rem('#logo');
  5.   //функция поп апа
  6.   $.popup('#area');
  7.   //скрыть
  8.   $.hide('#he');
  9.   //показать
  10.   $.show('#he');
  11.   //console.log
  12.   $.log(123);
  13.   //информация о эллементе
  14.   $.info('#he');
  15.   //изображение в консоль
  16.   $.logimg('img/logo.jpg');
  17.   //клик по эллемнту
  18.   $.click('.btn');
  19.   //изменяем текст эллемента
  20.   $.text('.btn', 'alert =3');
  21.   //добавляем класс
  22.   $.add('.btn', 'btnf');
  23.   //добавляем скрипт, стили
  24.   $.js('test.js');
  25.   $.css('test.css');
  26.   //заменяем {lib} на $lib
  27.   document.body.innerHTML = $.rnd({
  28.     lib: "$lib"
  29.   });
  30.   //добавляем input range
  31.   $.addR(
  32.     0,       //min range number
  33.     100,     //max range number
  34.     .5,      //range step
  35.     null,    //defolt number
  36.     null,    //if null, horizontal position
  37.     null,    //class, defold class="rng"
  38.     '#r'     //range position block, if null, document.body.appendChild
  39.   );

Примеры

$.rev - удаление

$.rem('#testrem');

$.hide и $.show - показать и скрыть

$.hide('#hideshow') и $.show('#hideshow')

$.popup - функция popup'а

Но проще делать это всё одной функцией

$.popup('#popup')

$.rnd - {info}

{p}

document.body.innerHTML = $.rnd({
  lib: "$lib"
});

$.addR - input range(beta)

Ползунок с настройками в одну строчку

$.addR(0, 100, .5, null, null, null, '#r');
